    class Solution {
    public TreeNode lowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
        if(root == null || root == p || root == q) return root;
        TreeNode left = lowestCommonAncestor(root.left, p, q);
        TreeNode right = lowestCommonAncestor(root.right, p, q);
        if(left == null && right == null) return null;
        if(left == null) return right;
        if(right == null) return left;
        return root;
    }
}

    # 51、寻找重复数

    class Solution {
    public int findDuplicate(int[] nums) {
        int n = nums.length - 1;
        int l = 1, r = n;
        //1到n,抽屉原理,二分个数
        while(l < r) {
            int mid = l + r >> 1;
            int cnt = 0;
            for(int num : nums) {
                if(num <= mid) cnt += 1;
            }
            if(cnt > mid) r = mid;
            else {
                l = mid + 1;
            }
        }
        return l;
    }
}

    # 52、最长上升子序列

    class Solution {
    static int N = 3100;
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[] f = new int[n + 1];
        int len = 0;
        int[] q = new int[N];
        for(int i = 0; i < n; i++) {
            int l = 0, r = len;
            while(l < r) {
                int mid = l + r + 1 >> 1;
                if(nums[i] <= q[mid]) {
                    r = mid - 1;
                }else{
                    l = mid;
                }
            }
            len = Math.max(len, r + 1);
            q[r + 1] = nums[i];
        }
        return len;
    }
}

    # 58、前k个高频元素

    //优先队列加map
class Solution {
    public int[] topKFrequent(int[] nums, int k) {
        int[] res = new int[k];
        Map<Integer, Integer> map = new HashMap<>();
        for(int num : nums) {
            map.put(num, map.getOrDefault(num, 0) + 1);
        }
        Set<Map.Entry<Integer, Integer>> set = map.entrySet();
        PriorityQueue<Map.Entry<Integer, Integer>> pri = new PriorityQueue<>((a, b) -> b.getValue() - a.getValue());
        for(Map.Entry<Integer, Integer> t : set) {
            pri.add(t);
        }
        for(int i = 0; i < k; i++) {
            res[i] = pri.poll().getKey();
        }
        return res;
    }
}

    # 59、字符串解码(dfs不一样)

    index[0]和index没搞明白.

    这里index是全局共享的字符串遍历下标，所以需要用数组来使得每层递归访问的index均是同一对象。这里也可以将index提取成 int成员变量，也能达到效果

    class Solution {
    public String decodeString(String s) {
            int[] index=new int[1];
            return analysis(s,index);
        }

    public String analysis(String s, int[] index){
        StringBuilder sb = new StringBuilder();
        int num=0;
        String temp;

        while(index[0]<s.length()){
            if(s.charAt(index[0])>='0'&&s.charAt(index[0])<='9'){
                num=num*10+s.charAt(index[0])-'0';
            }else if(s.charAt(index[0])=='['){
                index[0]++;
                temp = analysis(s,index);
                while(num-->0){
                    sb.append(temp);
                }
                num=0;
            }else if(s.charAt(index[0])==']'){
                break; //跳出递归返回结果
            }else{ 
                sb.append(s.charAt(index[0]));
            }
            // System.out.println(s.charAt(index[0]));
            index[0]++;
        }

        return sb.toString();
    }

}

    # 69、每日温度

    class Solution {
    public int[] dailyTemperatures(int[] temperatures) {
        int n = temperatures.length;
        Stack<Integer> st = new Stack<>();
        //单调递减栈
        int[] res = new int[n];
        for(int i = 0; i < n; i++) {
            while(!st.isEmpty() && temperatures[i] > temperatures[st.peek()]) {
                int t = st.pop();
                res[t] = i - t;
            }
            st.push(i);
        }
        return res;
    }
}
