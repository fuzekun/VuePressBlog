        int n = list.size();

            int a = list.get(x);

            for(int i = 2; i <= a / i; i++){
                if(a % i == 0){
                    int cnt = 0;

                    while(a % i == 0){
                        a /= i;
                        cnt++;
                    }
                    map.put(i, map.getOrDefault(i, 0) + cnt);

        int mod = (int)(1e9 + 7);
        for(int m : map.keySet()) {
            ans = (ans * (map.get(m) + 1)) % mod;
        }
        return ans;
    }

        // write code here
        return buildTree(preOrder, inOrder, 0, preOrder.size() - 1, 0, inOrder.size() - 1);

    public static ArrayList<TreeNode> buildTree(ArrayList<Integer> preOrder, ArrayList<Integer> inOrder, int pre_i, int pre_j, int in_i, int in_j) {
        ArrayList<TreeNode> res = new ArrayList<>();
        if (pre_i > pre_j) {
            res.add(null);
            return res;
        }
        int rootVal = preOrder.get(pre_i);
        ArrayList<Integer> candidates = new ArrayList<>();
        for (int i = in_i; i <= in_j; i++) {
            if (inOrder.get(i) == rootVal) {
                candidates.add(i);
            }
        }

        if (candidates.size() != 0) {
            for (int i : candidates) {
                int leftTreeSize = i - in_i;
                ArrayList<TreeNode> leftTrees = buildTree(preOrder, inOrder, pre_i + 1, pre_i + leftTreeSize, in_i, i - 1);
                ArrayList<TreeNode> rightTrees = buildTree(preOrder, inOrder, pre_i + leftTreeSize + 1, pre_j, i + 1, in_j);
                for (TreeNode lefTree : leftTrees) {
                    for (TreeNode rightTree : rightTrees) {
                        TreeNode root = new TreeNode(rootVal);
                        root.left = lefTree;
                        root.right = rightTree;
                        res.add(root);
                    }
                }
            }
        }
        return res;
    }

    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);

                i += 1;
            }else{
                i += 1;
            }
            ans = Math.max(ans, cnt);
        }
        System.out.println(ans);
    }
}

    import java.util.*;

public class Main {
    static int res = Integer.MAX_VALUE;
    public static void main(String[] args) {
        int[][] points = new int[3][2];
        points[0][0] = 0;
        points[0][1] = 1;
        points[1][0] = 0;
        points[1][1] = 2;
        points[2][0] = 0;
        points[2][1] = 3;
        System.out.println(getMinLen(points));
    }

    public static int getMinLen (int[][] points) {
        // write code here
        boolean[] visited = new boolean[points.length];
        backtrack(points,visited,0,0,0,0);
        return res;
    }
    public static void backtrack(int[][] points,boolean[] visited,int target,int length,int x,int y){
        if(target == points.length){
            res = Math.min(res,length);
            return;
        }
        for(int i = 0;i < points.length;i++){
            if(!visited[i]){
                length += Math.abs(points[i][0] - x) + Math.abs(points[i][1] - y);
                target++;
                visited[i] = true;
                backtrack(points,visited,target,length,points[i][0],points[i][1]);
                length -= Math.abs(points[i][0] - x) + Math.abs(points[i][1] - y);
                target--;
                visited[i] = false;
            }
        }
    }
}

class Main{    
    static List<List<Integer>> list;
    static boolean vis[];
    static long ret;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        vis = new boolean[n+1];
        ret = 0;
        list = new ArrayList<>();
        for(int i = 0; i <= n; i++){
            list.add(new ArrayList<>());
        }
        for(int i = 1; i < n; i++){
            int a = sc.nextInt();
            int b = sc.nextInt();

            list.get(b).add(a);

    public static void dfs(int node, int del) {
        if(node < del || ret == -1) {
            ret = -1;
            return;

        ret += node - del;
        vis[node] = true;
        for(int child : list.get(node)){
            if(!vis[child]){
                dfs(child, node);
            }
        }
    }
}

    // 给定一个只包含小写字母的字符串，求满足以下条件的子串数：

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        Map<Integer, Integer> hm = new HashMap<>();

        }
        System.out.println(res);
    }
}

    # 一、最美子字符串的数目 (opens new window)

    class Solution {
    public long wonderfulSubstrings(String word) {
        long[] cnt = new long[1 << 10];
        cnt[0] = 1; // 初始前缀和为 0，需将其计入出现次数
        long ans = 0L;
        for (int i = 0, sum = 0; i < word.length(); ++i) {
            sum ^= 1 << (word.charAt(i) - 'a'); // 计算当前前缀和
            ans += cnt[sum]; // 所有字母均出现偶数次
            for (int j = 0; j < 10; j ++) // 枚举其中一个字母出现奇数次
                ans += cnt[sum ^ (1 << j)]; // 反转该字母的出现次数的奇偶性
            ++cnt[sum]; // 更新前缀和出现次数
        }
        return ans;
    }
}

class Solution {
    public int maxProfit(int K, int[] prices) {
        int n = prices.length;
        if(prices.length == 0) return 0;
        //dp[i][j] ：第i天的状态为j，所剩下的最大现金是dp[i][j]
        int[][] dp = new int[n][2 * K + 1];
        for(int i = 1; i < 2 * K ; i+=2){
            dp[0][i] = -prices[0];
        }
        for(int i = 1; i < n; i++){
            for(int j = 0;j < 2 * K - 1; j += 2){
                //偶数是卖出，奇数是买入
                dp[i][j + 1] = Math.max(dp[i - 1][j + 1], dp[i - 1][j] - prices[i]);
                dp[i][j + 2] = Math.max(dp[i - 1][j + 2], dp[i - 1][j + 1] + prices[i]); 
            }
        }
        return dp[n - 1][2 * K];
    }
}

    # 三、从树中删除边的最小分数 (opens new window)

    class Solution {
    //先枚举删除一条边再dfs删除一条边
    int N = 1010;
    int[] h = new int[N];
    int[] e = new int[N * 2];
    int[] ne = new int[N * 2];
    int idx = 0;
    int[] w;
    int ans = Integer.MAX_VALUE;
    public void add(int a, int b){
        e[idx] = b;
        ne[idx] = h[a];
        h[a] = idx++;
    }
    
    public int dfs(int u, int fa, int sumx, int sumy){
        //返回的是异或值
        int res = w[u];
        for(int i = h[u]; i != -1; i = ne[i]){
            int j = e[i];
            if(j == fa) continue;//只向下走
            int t = dfs(j, u, sumx, sumy);
            res ^= t;
            //sumx, sumy, 
            if(sumx != -1){
                int[] a = new int[]{sumx ^ t, t, sumy};
                Arrays.sort(a);
                ans = Math.min(ans, a[2] - a[0]);
            }
        }
        return res;
    }
    public int minimumScore(int[] nums, int[][] edges) {
        w = nums;
        int n = nums.length;
        
        for(int i = 0;i < n - 1; i++){
            Arrays.fill(h, -1);//每次重新建图
            idx = 0;
            int a = edges[i][0];
            int b = edges[i][1];
            for(int j = 0; j < n - 1; j++){
                if(i != j){
                    int x = edges[j][0];
                    int y = edges[j][1];
                    add(x, y);
                    add(y, x);
                }
            }
            int sumx = dfs(a, -1, -1, -1);
            int sumy = dfs(b, -1, -1, -1);
            dfs(a, -1, sumx, sumy);
            dfs(b, -1, sumy, sumx);
        }
        return ans;
    }
}

    # 三、支配数

    • 思路： 滑动窗口：当窗口内任一元素达到 k 个时，窗口收缩。窗口收缩时，当前 子数组右边的 n - right 个元素逐一加入该子数组，都满足题目条件，所以统计时是加 n - right 而非加 1。
    import java.util.*;
class Test{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), k = sc.nextInt();
        int[] arr = new int[n];
        for(int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        int left = 0, right = 0;
        int res = 0;
        Map<Integer, Integer> map = new HashMap<>();
        while(right < n) {
            int num1 = arr[right];
            map.put(num1, map.getOrDefault(num1, 0) + 1);
            while(map.get(num1) == k) {
                int num2 = arr[left];
                left++;
                map.put(num2, map.get(num2) - 1);
                res += n - right;
            }
            right++;
        }
        System.out.println(res);
    }

}
